# 协程
无缓冲协程必须同时有收发方才能正常传输数据，否则阻塞

# 交替打印
使用三个协程，交替打印Dog,Fish,Cat,各100次

# 题解
1. 使用channel来控制输出顺序
2. 因为使用循环控制输出，最后一次Cat输出后继续向Dog管道发送控制量时，管道Dog此时不再接收，如果使用无缓冲通道会产生死锁，故使用缓冲通道

```golang
package main

import (
	"fmt"
	"sync"
)

var wg sync.WaitGroup
var count int = 100

func main() {

	dogCh, fishCh, catCh := make(chan struct{}, 1), make(chan struct{}), make(chan struct{})
	wg.Add(3)
	go Dog(dogCh, fishCh)
	go Fish(fishCh, catCh)
	go Cat(catCh, dogCh)
	//开始启动输出
	dogCh <- struct{}{}
	wg.Wait()
}

func Dog(dogCh, fishCh chan struct{}) {
	defer func() {
		wg.Done()
	}()
	for i := 1; i <= 100; i++ {
		<-dogCh
		fmt.Println("dog")
		fishCh <- struct{}{}
	}
}

func Fish(fishCh, catCh chan struct{}) {
	defer func() {
		wg.Done()
	}()
	for i := 1; i <= 100; i++ {
		<-fishCh
		fmt.Println("fish")
		catCh <- struct{}{}
	}

}

func Cat(catCh, dogCh chan struct{}) {
	defer func() {
		wg.Done()
	}()
	for i := 1; i <= 100; i++ {
		<-catCh
		fmt.Println("cat")
        //最后一次输出到dogCh为避免死锁出现，需要使用缓冲通道
		dogCh <- struct{}{}
	}
}
```
